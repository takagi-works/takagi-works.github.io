def transmit(packet, address, port)
  data = packet.is_a?(Takagi::Message::Outbound) ? packet.to_bytes : packet
  length = [data.bytesize].pack('n')
  socket = TCPSocket.new(address, port)
  socket.write(length + data)
  socket.close
end
