def run
  @shutdown = false
  trap('TERM') { @shutdown = true }
  trap('INT') { @shutdown = true }

  queue = Queue.new
  Array.new(@threads) { spawn_thread(queue) }

  @logger.debug "[Worker PID: #{Process.pid}] Listening on CoAP://0.0.0.0:#{@port} with #{@threads} threads"
  process_loop(queue)
end
